For the the first test I tried to upload Blink program

void setup() {
  // initialize digital pin LED_BUILTIN as an output.
  pinMode(13, OUTPUT);
}
// the loop function runs over and over again forever
void loop() {
  digitalWrite(13, HIGH);   // turn the LED on (HIGH is the voltage level)
  delay(1000);                       // wait for a second
  digitalWrite(13, LOW);    // turn the LED off by making the voltage LOW
  delay(1000);                       // wait for a second

void setup()
{
  // initialize digital pin 13 as an output.
  pinMode(13, OUTPUT);
  pinMode(10, OUTPUT);
  pinMode(9, OUTPUT);
  pinMode(7, OUTPUT);
}
 
void loop()
{
  for (int i = 0; i <= 3; i++) {
     switch (i) {
      case 0:
        digitalWrite(13, HIGH);
        digitalWrite(10, LOW);
        digitalWrite(9, LOW);
        digitalWrite(7, LOW);
        break;
      case 1:
        digitalWrite(13, LOW);
        digitalWrite(10, HIGH);
        digitalWrite(9, LOW);
        digitalWrite(7, LOW);
        break;
      case 2:
        digitalWrite(13, LOW);
        digitalWrite(10, LOW);
        digitalWrite(9, HIGH);
        digitalWrite(7, LOW);
        break;
      case 3:
        digitalWrite(13, LOW);
        digitalWrite(10, LOW);
        digitalWrite(9, LOW);
        digitalWrite(7, HIGH);
        break;
    }
    delay(250);
  }
}

Simplier version of LED run
void setup() {
  // initialize digital pin 13 as an output.
  pinMode(13, OUTPUT);
  pinMode(12, OUTPUT);
  pinMode(11, OUTPUT);
  pinMode(10, OUTPUT);
}


void loop() {

  for (int i = 13; i >= 10; i--) {
    digitalWrite(i, HIGH);
    delay(250);
    digitalWrite(13, LOW);
    digitalWrite(12, LOW);
    digitalWrite(11, LOW);
    digitalWrite(10, LOW);
  }

}
